select   Companies.Id,
         Companies.Name,
         Employees.Id,
         Employees.Age,
         Employees.CompanyId,
         Employees.Name
from     Companies as Companies
         left outer join
         Employees as Employees
         on Companies.Id = Employees.CompanyId
order by Companies.Name,
         Companies.Id
  }
